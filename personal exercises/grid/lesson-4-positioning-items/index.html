/* GRID COLUMN CONFIGURATION EXAMPLES: */

/* 1st APPROACH (Using Column Span): */
/* Sets the grid item to span 2 columns using the 'span' property */

/* 2nd APPROACH (Using Grid Lines): */

/* Method 1: Using 'grid-column-start' and 'grid-column-end' */
/* Sets the grid item to start at column line 1 and end at column line 3 */
grid-column-start: 1;
grid-column-end: 3;

/* Method 2: Using the shorthand 'grid-column: 1 / 3; */
grid-column: 1 / 3;

/* Method 3: Using the shorthand 'grid-column 1 / -1;' to span from the first to the last column line */
grid-column: 1 / -1;

}
